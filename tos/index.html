API Examples

Get started with these code snippets. Refer to our docs for complete endpoint details.

Fetch User Data (JavaScript)

fetch('https://api.jeromegpt.my.id/user')
  .then(response => response.json())
  .then(data => console.log(data));

Post Comment (Python)

import requests
response = requests.post('https://api.jeromegpt.my.id/comment', json={'text': 'Hello!'})
print(response.json())

Search Query (Node.js)

const axios = require('axios');
axios.get('https://api.jeromegpt.my.id/search?q=query')
  .then(response => console.log(response.data))
  .catch(error => console.error(error));

AI Query (Go)

package main
import ("fmt"; "net/http"; "io/ioutil")
func main() {
  resp, err := http.Get("https://api.jeromegpt.my.id/ai?q=hello")
  if err != nil { fmt.Println(err); return }
  defer resp.Body.Close()
  body, err := ioutil.ReadAll(resp.Body)
  if err != nil { fmt.Println(err); return }
  fmt.Println(string(body))
}
